## 📋 РЕЗЮМЕ

На основе анализа текущей архитектуры (`desktop.html`, `wizard.js`, `weather.js`, `energy-model.js`) и технического задания (`DASHBOARD-PROMPT.md`) разработан план трансформации системы.

## 🔍 АНАЛИЗ ТЕКУЩЕГО СОСТОЯНИЯ

### 1. Архитектурные проблемы

| Проблема | Описание | Влияние |
|----------|----------|---------|
| **Монолитный wizard.js** | 4127 строк, 67+ функций | Сложность тестирования, поддержки |
| **Смешанная ответственность** | UI-логика + бизнес-логика в одном модуле | Нарушение SRP |
| **Глобальное состояние** | Данные разбросаны по модулям | Рассинхронизация, race conditions |
| **Отсутствие единого хранилища** | Каждый модуль хранит данные по-своему | Дублирование, inconsistency |
| **Жёсткая связность** | Модули зависят от DOM-элементов | Невозможность unit-тестирования |
| **Нет обработки ошибок** | API failures не централизованы | Плохой UX при сбоях |

### 2. Визуальные проблемы

| Проблема | Описание |
|----------|----------|
| **50%/50% разделение** | Неэффективное использование пространства |
| **Полноэкранный режим** | Костыльное решение через CSS классы |
| **Отсутствие адаптивности** | Фиксированные размеры для десктопа |
| **Перегруженная шапка** | 5 кнопок в `panel-actions` |
| **Неясная иерархия** | Все элементы визуально равны |

### 3. UX проблемы

| Проблема | Описание |
|----------|----------|
| **4 шага обязательны** | Нельзя перейти к шагу 3 без шага 1 |
| **Нет состояния "загрузки"** | Пользователь не видит прогресс |
| **Отсутствие подсказок** | Неясно, что делать дальше |
| **Нет истории действий** | Нельзя отменить действие |
| **Сложный экспорт** | PDF генерируется без предпросмотра |

### 1. Принципы проектирования

```
┌─────────────────────────────────────────────────────────┐
│                    MIRA v0.2.0                          │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  ┌─────────────┐    ┌──────────────┐    ┌────────────┐ │

│  └──────┬──────┘    └──────────────┘    └────────────┘ │
│         │                                               │
```

**Принципы:**
- ✅ **Single Source of Truth** — единое хранилище состояния
- ✅ **Unidirectional Data Flow** — данные текут в одном направлении
- ✅ **Separation of Concerns** — разделение логики и представления
- ✅ **Immutability** — состояние изменяется только через действия
- ✅ **Composability** — компоненты переиспользуемы

## 📊 МОДУЛЬНАЯ ДЕКОМПОЗИЦИЯ

### 1. Store (Хранилище состояния)

```javascript
// js/core/store.js

const initialState = {
    // Маршрут
    route: {
        points: [],           // Точки маршрута [A, B, C...]
        segments: [],         // Разбиение на сегменты
        distance: 0,          // Общая дистанция (км)
        flightTime: 0,        // Время полёта (мин)
        status: 'idle'        // idle | loading | success | error
    },

    // Метеоданные
    weather: {
        forecast: null,       // Сырые данные API
        analyzed: null,       // Проанализированные данные
        segments: [],         // Данные по сегментам
        riskLevel: null,      // Общий уровень риска
        status: 'idle'
    },

    // Энергия
    energy: {
        profile: null,        // Профиль БВС
        calculation: null,    // Результат расчёта
        balance: null,        // Баланс "Туда/Обратно"
        status: 'idle'
    },

    // Пилот
    pilot: {
        observations: [],     // Наблюдения (ground/flight)
        correction: null,     // Применённая коррекция
        coords: null,         // Координаты точки старта
        status: 'idle'
    },

    // UI
    ui: {
        currentTab: 'meteo',  // Активная вкладка дашборда
        fullscreen: false,    // Полноэкранный режим
        loading: false,       // Глобальный индикатор загрузки
        notifications: []     // Уведомления
    }
};

class Store {
    constructor() {
        this.state = { ...initialState };
        this.listeners = [];
    }

    getState() { return this.state; }

    setState(newState) {
        this.state = { ...this.state, ...newState };
        this.notify();
    }

    subscribe(listener) {
        this.listeners.push(listener);
        return () => {
            this.listeners = this.listeners.filter(l => l !== listener);
        };
    }

    notify() {
        this.listeners.forEach(listener => listener(this.state));
    }
}

export const store = new Store();
```

### 3. DashboardModule (новый модуль)

```javascript
// js/modules/dashboard/dashboard.js

const DashboardModule = {
    isOpen: false,
    activeTab: 'meteo',

    tabs: [
        { id: 'meteo',    label: 'Метеопрогноз',  icon: 'fa-cloud-sun' },
        { id: 'ground',   label: 'Сидя на земле', icon: 'fa-flag' },
        { id: 'flight',   label: 'В полёте',      icon: 'fa-plane' },
        { id: 'segments', label: 'Сегменты',      icon: 'fa-map' },
        { id: 'energy',   label: 'Энергия',       icon: 'fa-battery-three-quarters' },
        { id: 'report',   label: 'Отчёт',         icon: 'fa-file-pdf' }
    ],

    /**
     * Инициализация дашборда
     */
    init() {
        this.renderModal();
        this.bindEvents();
        this.updateButtonState();
    },

    /**
     * Отрисовка модального окна
     */
    renderModal() {
        const html = `
            <div id="dashboardModal" class="dashboard-modal">
                <div class="modal-header">
                    <h1><i class="fas fa-tachometer-alt"></i> MIRA Dashboard</h1>
                    <button class="close-btn" onclick="DashboardModule.close()">
                        <i class="fas fa-times"></i>
                    </button>
                </div>

                <div class="modal-tabs">
                    ${this.tabs.map(tab => `
                        <button class="modal-tab" data-tab="${tab.id}">
                            <i class="fas ${tab.icon}"></i>
                            <span>${tab.label}</span>
                        </button>
                    `).join('')}
                </div>

                <div class="modal-body">
                    <div id="tab-meteo" class="tab-content"></div>
                    <div id="tab-ground" class="tab-content"></div>
                    <div id="tab-flight" class="tab-content"></div>
                    <div id="tab-segments" class="tab-content"></div>
                    <div id="tab-energy" class="tab-content"></div>
                    <div id="tab-report" class="tab-content"></div>
                </div>

                <div class="modal-footer">
                    <button class="back-btn" onclick="DashboardModule.close()">
                        <i class="fas fa-arrow-left"></i> Назад к карте
                    </button>
                </div>
            </div>
        `;

        document.body.insertAdjacentHTML('beforeend', html);
    },

    /**
     * Открытие дашборда
     */
    open() {
        const modal = document.getElementById('dashboardModal');
        modal.classList.add('active');
        this.isOpen = true;
        this.switchTab(this.activeTab);

        // Блокировка прокрутки body
        document.body.style.overflow = 'hidden';

        // Фокус на первой доступной вкладке
        this.focusFirstAvailableTab();
    },

    /**
     * Закрытие дашборда
     */
    close() {
        const modal = document.getElementById('dashboardModal');
        modal.classList.remove('active');
        this.isOpen = false;

        // Восстановление прокрутки body
        document.body.style.overflow = '';
    },

    /**
     * Переключение вкладки
     */
    switchTab(tabId) {
        this.activeTab = tabId;

        // Обновление UI вкладок
        document.querySelectorAll('.modal-tab').forEach(tab => {
            tab.classList.toggle('active', tab.dataset.tab === tabId);
        });

        // Скрытие всех вкладок
        document.querySelectorAll('.tab-content').forEach(content => {
            content.style.display = 'none';
        });

        // Показ активной вкладки
        const activeContent = document.getElementById(`tab-${tabId}`);
        activeContent.style.display = 'block';

        // Рендер контента вкладки
        this.renderTabContent(tabId);
    },

    /**
     * Проверка доступности вкладки
     */
    isTabAvailable(tabId) {
        const state = store.getState();

        switch (tabId) {
            case 'meteo':
                return state.weather.forecast !== null || state.route.segments.length > 0;
            case 'ground':
                return state.pilot.observations.filter(o => o.type === 'ground').length > 0;
            case 'flight':
                return state.pilot.observations.filter(o => o.type === 'flight').length > 0;
            case 'segments':
                return state.route.segments.length > 0;
            case 'energy':
                return state.energy.calculation !== null;
            case 'report':
                return true;  // Всегда доступна
        }
    },

    /**
     * Обновление состояния кнопки
     */
    updateButtonState() {
        const button = document.getElementById('dashboardButton');
        if (!button) return;

        const state = store.getState();
        const hasData =
            state.weather.forecast !== null ||
            state.pilot.observations.length > 0 ||
            state.route.segments.length > 0;

        if (hasData) {
            button.disabled = false;
            button.classList.add('active');
            button.classList.remove('disabled');
        } else {
            button.disabled = true;
            button.classList.remove('active');
            button.classList.add('disabled');
        }
    },

    /**
     * Рендер контента вкладки
     */
    renderTabContent(tabId) {
        switch (tabId) {
            case 'meteo':
                MeteoTabModule.render();
                break;
            case 'ground':
                GroundTabModule.render();
                break;
            case 'flight':
                FlightTabModule.render();
                break;
            case 'segments':
                SegmentsTabModule.render();
                break;
            case 'energy':
                EnergyTabModule.render();
                break;
            case 'report':
                ReportTabModule.render();
                break;
        }
    },

    /**
     * Привязка событий
     */
    bindEvents() {
        // Переключение вкладок
        document.getElementById('dashboardModal').addEventListener('click', (e) => {
            const tab = e.target.closest('.modal-tab');
            if (tab && this.isTabAvailable(tab.dataset.tab)) {
                this.switchTab(tab.dataset.tab);
            }
        });

        // Закрытие по Esc
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape' && this.isOpen) {
                this.close();
            }
        });
    },

    /**
     * Фокус на первой доступной вкладке
     */
    focusFirstAvailableTab() {
        for (const tab of this.tabs) {
            if (this.isTabAvailable(tab.id)) {
                this.switchTab(tab.id);
                break;
            }
        }
    }
};
```
